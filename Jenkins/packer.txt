pipeline {
    agent any
    
    environment {
        AWS_ACCESS_KEY_ID = credentials('AWS_ACCESS_KEY_ID')
        AWS_SECRET_ACCESS_KEY = credentials('AWS_SECRET_ACCESS_KEY')
        AWS_DEFAULT_REGION = 'eu-west-1'
    }
    
    stages {
        stage ('Checkout'){
            steps {
                git branch : 'main',
                url : 'https://github.com/oyediran-phillip-oyeniyi/Techbleat-week-6.git'
            }
        }
        stage ( 'Packer Initialization'){
            steps {
                dir ('packer'){
                    sh 'packer init .'
                }
            }
        }
        stage ( 'Packer Validateion'){
            steps {
                dir ('packer'){
                    sh 'packer validate .'
                }
            }
        }
        stage ( 'Packer Building'){
            steps {
                dir ('packer'){
                    sh 'packer build .'
                }
            }
        }
        
    }
}
